"""
QRGB - Generador de C√≥digos QR Multicapa
Desarrollado por Federico Anderson con asistencia de Claude Sonnet 3.5
"""

import qrcode
import numpy as np
from PIL import Image

class QRGBGenerator:
    """Generador de c√≥digos QR con m√∫ltiples capas de informaci√≥n en RGB"""
    
    def __init__(self):
        """Inicializa el generador de c√≥digos QRGB"""
        self.qr_version = 1  # Versi√≥n predeterminada de QR
        self.error_correction = qrcode.constants.ERROR_CORRECT_M  # Nivel de correcci√≥n de errores
    
    def set_version(self, version):
        """Establece la versi√≥n del c√≥digo QR (1-40)"""
        if 1 <= version <= 40:
            self.qr_version = version
        else:
            raise ValueError("La versi√≥n debe estar entre 1 y 40")
    
    def set_error_correction(self, level):
        """Establece el nivel de correcci√≥n de errores"""
        levels = {
            'L': qrcode.constants.ERROR_CORRECT_L,  # ~7% de correcci√≥n
            'M': qrcode.constants.ERROR_CORRECT_M,  # ~15% de correcci√≥n
            'Q': qrcode.constants.ERROR_CORRECT_Q,  # ~25% de correcci√≥n
            'H': qrcode.constants.ERROR_CORRECT_H   # ~30% de correcci√≥n
        }
        if level in levels:
            self.error_correction = levels[level]
        else:
            raise ValueError("Nivel de correcci√≥n no v√°lido. Use 'L', 'M', 'Q' o 'H'")
    
    def generate_qr_layer(self, data, color=(0, 0, 0)):
        """Genera una capa de c√≥digo QR para un canal de color espec√≠fico"""
        qr = qrcode.QRCode(
            version=self.qr_version,
            error_correction=self.error_correction,
            box_size=10,
            border=4,
        )
        qr.add_data(data)
        qr.make(fit=True)
        
        # Crear imagen QR en el canal especificado
        img = qr.make_image(fill_color=color, back_color="white")
        return img
    
    def generate_rgb_qr(self, red_data, green_data, blue_data):
        """Genera un c√≥digo QR multicapa combinando tres mensajes en los canales RGB"""
        # Generar c√≥digos QR para cada canal
        red_qr = self.generate_qr_layer(red_data, color=(255, 0, 0))
        green_qr = self.generate_qr_layer(green_data, color=(0, 255, 0))
        blue_qr = self.generate_qr_layer(blue_data, color=(0, 0, 255))
        
        # Convertir a arrays de numpy
        red_array = np.array(red_qr)
        green_array = np.array(green_qr)
        blue_array = np.array(blue_qr)
        
        # Asegurar que todas las im√°genes tienen el mismo tama√±o
        min_height = min(red_array.shape[0], green_array.shape[0], blue_array.shape[0])
        min_width = min(red_array.shape[1], green_array.shape[1], blue_array.shape[1])
        
        # Recortar al tama√±o m√≠nimo
        red_array = red_array[:min_height, :min_width]
        green_array = green_array[:min_height, :min_width]
        blue_array = blue_array[:min_height, :min_width]
        
        # Combinar los canales
        rgb_array = np.zeros((min_height, min_width, 3), dtype=np.uint8)
        
        # Establecer los canales R, G y B basados en los c√≥digos QR
        rgb_array[..., 0] = red_array[..., 0]  # Canal rojo
        rgb_array[..., 1] = green_array[..., 1]  # Canal verde
        rgb_array[..., 2] = blue_array[..., 2]  # Canal azul
        
        # Crear imagen final
        combined_img = Image.fromarray(rgb_array)
        return combined_img
    
    def save_qr(self, image, filename):
        """Guarda la imagen QR en un archivo"""
        image.save(filename)
        return filename
